\subsection{模型求解代码}

\begin{lstlisting}[language=Python, caption=模型求解代码]
from scipy.optimize import minimize

# 目标函数
def objective(x):
    return x[0]**2 + x[1]**2

# 约束条件
def constraint(x):
    return x[0] + x[1] - 1

# 求解
x0 = [0, 0]
result = minimize(objective, x0, constraints={'type': 'eq', 'fun': constraint})

print("最优解:", result.x)
print("最优值:", result.fun)
\end{lstlisting}
